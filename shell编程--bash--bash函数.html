使用$*传参
[root@wing shell]# cat a.sh
#!/bin/bash
a=(1 2 3 4 5)
hello(){
for i in $* #因为是一个数组,不知道有多少个参数,所以使用$*获取所有参数
do
echo $[$i*2]
done
}
hello ${a[*]}
